def self.find_by_name(name)
  disk = GameObj.loot.find do |item|
    is_disk?(item) && item.name.include?(name)
  end
  return nil if disk.nil?
  Disk.new(disk)
end

def self.is_disk?(thing)
  thing.name =~ /\b([A-Z][a-z]+) #{Regexp.union(NOUNS)}\b/
end
