def self.create(game_type)
  case game_type
  when /^GS/
    Gemstone::GameInstance.new
  when /^DR/
    DragonRealms::GameInstance.new
  else
    # Default to a basic implementation if game type is unknown
    GameInstance::Base.new
  end
end
